import requests

overpass_url = "http://overpass-api.de/api/interpreter"
overpass_query = """
    [out:json];
    area["ISO3166-1"="GH"][admin_level=2];
    (
    node["amenity"="school"](area);
    );
    out count;"""
response = requests.get(overpass_url, params={'data': overpass_query})
response.json()
